  x)

(declare t-dtg28-good (T-dtg28 :a -> Integer))
(define (t-dtg28-good g)
  (match g
    ((= whole (T28-Int x))
     (+ x (t-dtg28-require-int whole)))
    ((T28-Str s)
     (as Integer (coalton/string:length s)))))
